代碼示例

// HalfShader示例代碼
#include "HalfShader.h"

int main() {
    HalfShader::Compiler compiler;
    
    // 加載著色器源代碼
    std::string vertexShaderSource = LoadShaderSource("vertex.glsl");
    std::string fragmentShaderSource = LoadShaderSource("fragment.glsl");
    
    // 設置編譯選項
    HalfShader::CompileOptions options;
    options.targetAPI = HalfShader::API::Vulkan;
    options.optimizationLevel = HalfShader::OptimizationLevel::High;
    
    // 編譯著色器
    HalfShader::CompiledShader vertexShader = compiler.Compile(vertexShaderSource, HalfShader::ShaderType::Vertex, options);
    HalfShader::CompiledShader fragmentShader = compiler.Compile(fragmentShaderSource, HalfShader::ShaderType::Fragment, options);
    
    // 檢查編譯結果
    if (vertexShader.IsValid() && fragmentShader.IsValid()) {
        // 使用編譯後的著色器
        UseCompiledShaders(vertexShader, fragmentShader);
    } else {
        // 處理編譯錯誤
        std::cout << "Shader compilation failed: " << compiler.GetLastError() << std::endl;
    }
    
    return 0;
}
