代碼示例


#include <HalfAudio/HalfAudio.h>

// 初始化音頻系統
HalfAudio::AudioSystem audioSystem;
audioSystem.initialize();

// 加載音頻文件
HalfAudio::SoundPtr backgroundMusic = audioSystem.loadSound("background.ogg");
HalfAudio::SoundPtr explosionSound = audioSystem.loadSound("explosion.wav");

// 播放背景音樂（循環播放）
HalfAudio::ChannelPtr musicChannel = audioSystem.playSound(backgroundMusic, true);

// 設置3D音頻源
HalfAudio::AudioSource explosionSource;
explosionSource.setPosition(10.0f, 0.0f, 5.0f);

// 播放3D音效
audioSystem.playSound3D(explosionSound, explosionSource);

// 應用音頻特效
HalfAudio::Effect reverbEffect = HalfAudio::createReverbEffect();
reverbEffect.setDecayTime(1.5f);
musicChannel->addEffect(reverbEffect);

// 動態調整音量
musicChannel->setVolume(0.8f);

// 清理資源
audioSystem.shutdown();
