代碼示例


// HalfScript示例代碼

// 定義一個簡單的Player類
class Player {
    var name: string
    var health: int
    var score: int

    constructor(name: string) {
        self.name = name
        self.health = 100
        self.score = 0
    }

    func takeDamage(amount: int) {
        self.health -= amount
        if self.health < 0 {
            self.health = 0
        }
    }

    func addScore(points: int) {
        self.score += points
    }
}

// 創建玩家實例
var player = Player("半人")

// 游戲邏輯
func updateGame() {
    player.takeDamage(10)
    player.addScore(50)
    print("玩家 {player.name} 的生命值: {player.health}, 得分: {player.score}")
}

// 主循環
while player.health > 0 {
    updateGame()
    yield // 協程讓出控制權
}

print("游戲結束！")
